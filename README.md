```py
#%% 자연수를 한글로 변경
# 입력 예) 1024
# 출력 예) 일공이사
han = "공일이삼사오육칠팔구"
num = int(input())
result = ""

while num != 0:
    result = han[num%10] + result
    num //=10

print(result)
```
